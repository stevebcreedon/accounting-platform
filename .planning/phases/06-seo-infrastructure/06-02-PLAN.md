1. **Create `src/app/sitemap.ts`** — Dynamic XML sitemap using Next.js built-in convention:
   ```typescript
   import type { MetadataRoute } from 'next';
   import { getAllGuides } from '@/lib/content/queries';
   import { getAllCategorySlugs } from '@/lib/content/categories';

   const BASE_URL = process.env.NEXT_PUBLIC_SITE_URL || 'https://the-ledger.vercel.app';

   export default function sitemap(): MetadataRoute.Sitemap {
     const guides = getAllGuides().map((guide) => ({
       url: `${BASE_URL}/guides/${guide.slug}`,
       lastModified: guide.updatedDate || guide.publishDate,
       changeFrequency: 'monthly' as const,
       priority: guide.isPillar ? 0.9 : 0.7,
     }));

     const categories = getAllCategorySlugs().map((slug) => ({
       url: `${BASE_URL}/guides/category/${slug}`,
       lastModified: new Date().toISOString(),
       changeFrequency: 'weekly' as const,
       priority: 0.6,
     }));

     const staticPages: MetadataRoute.Sitemap = [
       { url: BASE_URL, priority: 1.0, changeFrequency: 'daily', lastModified: new Date().toISOString() },
       { url: `${BASE_URL}/guides`, priority: 0.8, changeFrequency: 'daily', lastModified: new Date().toISOString() },
       { url: `${BASE_URL}/find-accountant`, priority: 0.6, changeFrequency: 'monthly', lastModified: new Date().toISOString() },
       { url: `${BASE_URL}/about`, priority: 0.5, changeFrequency: 'monthly', lastModified: new Date().toISOString() },
       { url: `${BASE_URL}/contact`, priority: 0.4, changeFrequency: 'monthly', lastModified: new Date().toISOString() },
       { url: `${BASE_URL}/privacy`, priority: 0.2, changeFrequency: 'yearly', lastModified: new Date().toISOString() },
       { url: `${BASE_URL}/terms`, priority: 0.2, changeFrequency: 'yearly', lastModified: new Date().toISOString() },
     ];

     return [...staticPages, ...guides, ...categories];
   }
   ```

   This lists: 7 static pages + 3 guide articles + 8 category pages = 18 URLs. No sitemap index needed for this count (D-10).

2. **Create `src/app/robots.ts`** — Allow all crawlers per D-11:
   ```typescript
   import type { MetadataRoute } from 'next';

   const BASE_URL = process.env.NEXT_PUBLIC_SITE_URL || 'https://the-ledger.vercel.app';

   export default function robots(): MetadataRoute.Robots {
     return {
       rules: [
         {
           userAgent: '*',
           allow: '/',
         },
       ],
       sitemap: `${BASE_URL}/sitemap.xml`,
     };
   }
   ```

   Per D-11: No disallow rules for any crawler. GPTBot, CCBot, Googlebot, and all others are allowed. This maximizes content visibility for both traditional search and AI search engines.

IMPORTANT: Use the same BASE_URL fallback value ('https://the-ledger.vercel.app') as used in schemas.ts for consistency. Import from schemas.ts if preferred, or use the env var directly.
