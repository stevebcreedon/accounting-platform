2. Rewrite `src/app/layout.tsx` with the following exact content:

   ```typescript
   import type { Metadata, Viewport } from 'next';
   import { DM_Sans } from 'next/font/google';
   import localFont from 'next/font/local';
   import './globals.css';

   const dmSans = DM_Sans({
     subsets: ['latin'],
     variable: '--font-dm-sans',
     display: 'swap',
     weight: ['400', '500', '700'],
   });

   const satoshi = localFont({
     src: [
       {
         path: '../../public/fonts/Satoshi-Variable.woff2',
         style: 'normal',
       },
       {
         path: '../../public/fonts/Satoshi-VariableItalic.woff2',
         style: 'italic',
       },
     ],
     variable: '--font-satoshi',
     display: 'swap',
   });

   export const metadata: Metadata = {
     title: {
       default: 'The Ledger — Irish Accounting Guidance',
       template: '%s | The Ledger',
     },
     description:
       'Clear, jargon-free accounting guidance for Irish small businesses, sole traders, and company directors.',
   };

   export const viewport: Viewport = {
     width: 'device-width',
     initialScale: 1,
     themeColor: '#FAF8F5',
   };

   export default function RootLayout({
     children,
   }: {
     children: React.ReactNode;
   }) {
     return (
       <html lang="en" className={`${dmSans.variable} ${satoshi.variable}`}>
         <body className="font-body bg-cream text-charcoal antialiased">
           {children}
         </body>
       </html>
     );
   }
   ```

   Key details:
   - DM_Sans imported from `next/font/google` with weights `['400', '500', '700']` (per UI-SPEC: only 400 and 700 used, 500 for flexibility)
   - Satoshi loaded via `next/font/local` from `public/fonts/` woff2 files
   - Both fonts expose CSS variables: `--font-dm-sans` and `--font-satoshi`
   - Both variables applied to `<html>` element className
   - `<body>` uses `font-body` (maps to DM Sans via Tailwind), `bg-cream` (#FAF8F5), `text-charcoal` (#1C1917)
   - `lang="en"` set on html element
   - Metadata uses "The Ledger" brand name per D-11
   - Viewport includes themeColor matching cream background
