3. Create `src/lib/rate-limit.ts` -- in-memory IP-based rate limiter (per D-10, research Pattern 4):
   ```typescript
   const requests = new Map<string, { count: number; resetTime: number }>();

   export function rateLimit(ip: string, limit = 5, windowMs = 60_000): boolean {
     const now = Date.now();
     const entry = requests.get(ip);

     if (!entry || now > entry.resetTime) {
       requests.set(ip, { count: 1, resetTime: now + windowMs });
       return true;
     }

     if (entry.count >= limit) {
       return false;
     }

     entry.count++;
     return true;
   }
   ```
   Returns `true` if allowed, `false` if rate limited. Resets on serverless cold start -- acceptable per D-10.

   CREATE TABLE IF NOT EXISTS public.subscribers (
     id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
     email TEXT NOT NULL UNIQUE,
     source TEXT NOT NULL CHECK (source IN ('homepage', 'directory-waitlist', 'article-cta')),
     confirmed BOOLEAN DEFAULT FALSE,
